```typescript
interface PackageFeatures {
```

### Feature Check Implementation

```typescript
// hooks/usePackageRestrictions.ts
export const usePackageRestrictions = () => {
    const { user } = useAuth();
    const packageFeatures = getPackageFeatures(user.plan);
    
    const canViewProfile = (profileType: 'advocate' | 'featured') => {
        if (profileType === 'advocate') {
            return packageFeatures.profileVisibility;
        }
        return packageFeatures.featuredProfileAccess;
    };
    
    const shouldShowStarMarks = () => {
        return packageFeatures.starMarkPercentage > 0;
    };
    
    const getStarMarkPercentage = () => {
        return packageFeatures.starMarkPercentage;
    };
    
    const shouldShowUpgradePrompt = (feature: string) => {
        return packageFeatures.showUpgradePrompts && !hasFeatureAccess(feature);
    };
    
    return {
        canViewProfile,
        shouldShowStarMarks,
        getStarMarkPercentage,
        shouldShowUpgradePrompt,
        features: packageFeatures
    };
};
```

```tsx
// components/AdvocateProfile.tsx
const AdvocateProfile = ({ advocate }) => {
    const { canViewProfile, getStarMarkPercentage, shouldShowUpgradePrompt } = usePackageRestrictions();
    
    const visibilityPercentage = canViewProfile('advocate');
    const starMarkPercentage = getStarMarkPercentage();
    
    // Ultra Pro: 100% visible, no restrictions
    if (visibilityPercentage === 100) {
        return <FullAdvocateProfile advocate={advocate} />;
    }
    
    // Partial visibility with star marks
    return (
        <div className={styles.profileContainer}>
            {/* Visible content */}
            <div style={{ opacity: visibilityPercentage / 100 }}>
                <AdvocateBasicInfo advocate={advocate} />
            </div>
            
            {/* Star marks overlay */}
            {starMarkPercentage > 0 && (
                <div className={styles.starMarkOverlay}>
                    <div className={styles.stars}>
                        {Array(Math.floor(starMarkPercentage / 10)).fill('â­').map((star, i) => (
                            <span key={i}>{star}</span>
                        ))}
                    </div>
                    {shouldShowUpgradePrompt('full_profile_access') && (
                        <UpgradePrompt feature="View full advocate profiles" />
                    )}
                </div>
            )}
        </div>
    );
};
```

### 2. Featured Profile Blur

```tsx
// components/FeaturedProfile.tsx
const FeaturedProfile = ({ profile }) => {
    const { canViewProfile } = usePackageRestrictions();
    
    const accessLevel = canViewProfile('featured');
    
    // Ultra Pro: Full access
    if (accessLevel === 100) {
        return <FullFeaturedProfile profile={profile} />;
    }
    
    // Free users: Completely blurred
    if (accessLevel < 30) {
        return (
            <div className={styles.blurredProfile}>
                <div className={styles.blurOverlay}>
                    <Lock size={48} />
                    <h3>Premium Feature</h3>
                    <p>Upgrade to view featured profiles</p>
                    <UpgradeButton />
                </div>
            </div>
        );
    }
    
    // Partial access: Some blur
    return (
        <div className={styles.partialProfile} style={{ filter: `blur(${(100 - accessLevel) / 20}px)` }}>
            <FeaturedProfileContent profile={profile} />
            <UpgradeOverlay accessLevel={accessLevel} />
        </div>
    );
};
```
